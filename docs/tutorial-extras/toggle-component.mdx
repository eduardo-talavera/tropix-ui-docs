# Toggle

## Toggle props
- `variant: ` → `string options: 'jungleGreen' | 'cobaltBlue' | 'poppy' | 'sunGlow'`
- `size: ` → `string options: 'sm' | 'md' | 'lg'`
- `isThemeSwitch` → `boolean`
- `isToggled` → `boolean`
- `onPress` → `(toggled: boolean) => void`




## isThemeSwitch and onPress props

```jsx
import { Toggle } from 'tropix-ui'


export const App = () => {

  const [isToggled, setIsToggled] = useState(false)
   
  return <>
      <Toggle 
        isThemeSwitch
        onPress={(toggled) => setIsToggled(toggled)}
      />
      
      <p>{ isToggled ? 'Toggled On' : 'Toggled Off' }</p>
  </>
};
```

## Variant and isToggled props

```jsx
import { Toggle } from "tropix-ui"

const toggleVariants = ['jungleGreen', 'cobaltBlue', 'poppy', 'sunGlow']

export const ToggleVariants = () => {
  return (
    <>
      {
        toggleVariants.map(variant => (
            <Toggle
                key={variant}
                variant={variant}
                style={{ margin: '1rem' }}
                onPress={() => {}}
                isToggled
            />
        ))
      }
    </>
  )
}
```
